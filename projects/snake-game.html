Neon Snake

A classic arcade game rebuilt with HTML5 Canvas. Features array-based movement logic and collision detection.

// --- Game Loop: Update Logic ---
// Runs 10 times per second

function update() {
    // 1. Calculate new head position
    const head = { 
        x: snake[0].x + dx, 
        y: snake[0].y + dy 
    };

    // 2. Check for Wall Collision
    if (head.x < 0 || head.x >= TILE_COUNT || 
        head.y < 0 || head.y >= TILE_COUNT) {
        gameOver();
        return;
    }

    // 3. Move Snake (Add Head)
    snake.unshift(head);

    // 4. Eat Food logic
    if (head.x === food.x && head.y === food.y) {
        score += 10;
        placeFood();
        // Don't pop tail (snake grows)
    } else {
        snake.pop(); // Remove tail to maintain size
    }
}
