Neon RPG Engine

A tile-based RPG prototype featuring grid movement, collision detection, random encounters, and a turn-based battle system with leveling logic.

// --- Game State & Movement Logic ---
// Controls player exploration and triggers battles

let gameState = 'EXPLORING'; // or 'BATTLING'

function movePlayer(dx, dy) {
    // 1. Prevent movement during battle
    if (gameState !== 'EXPLORING') return;

    const newX = player.x + dx;
    const newY = player.y + dy;

    // 2. Collision Detection against Map Walls (Tile ID 1)
    if (newX >= 0 && newX < COLS && 
        newY >= 0 && newY < ROWS && 
        map[newY][newX] !== 1) {
        
        // Valid move: Update position
        player.x = newX;
        player.y = newY;
        
        // 3. Check for Random Encounter
        if (Math.random() < ENCOUNTER_CHANCE) {
            startBattle(); // Switches gameState to 'BATTLING'
        }
    }
}
