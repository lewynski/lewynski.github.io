Toy Kingdom

A physics-based claw machine game. Features collision detection and a "Slippage" difficulty mechanic that randomly drops items in Hard Mode.

// --- Difficulty Logic: The "Slip" ---
// In Hard Mode, the claw might drop the prize!

if (claw.state === 'RETRACTING') {
    
    // 1. Pull claw up
    claw.y -= CLAW_SPEED_Y;
    
    // 2. Check if holding item
    if (claw.grabbedItem) {
        
        // 3. RNG Check for Slippage
        if (difficulty === 'HARD' && 
            Math.random() < SLIP_CHANCE) {
            
            // OOPS! Drop the item back to floor
            claw.grabbedItem.y = FLOOR_Y;
            prizes.push(claw.grabbedItem);
            
            // Reset Claw Grip
            claw.grabbedItem = null; 
        }
    }
}
