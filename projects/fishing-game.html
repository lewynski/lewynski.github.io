Neon Angler

A physics-lite arcade game featuring a 3-stage state machine (Idle, Casting, Reeling) and object-based collision detection.

// --- State Machine & Collision ---
// Handles the hook logic frame-by-frame

if (hook.state === 'CASTING') {
    // 1. Move Hook Down
    hook.y += hook.speed;
    
    // 2. Check Hit with Fish
    fishList.forEach(f => {
        if (!f.caught && 
           Math.abs(hook.x - f.x) < 20 && 
           Math.abs(hook.y - f.y) < 20) {
            
            // STATE CHANGE: Hit!
            hook.state = 'REELING';
            hook.target = f;
            f.caught = true;
        }
    });

} else if (hook.state === 'REELING') {
    // 3. Drag Fish Up
    hook.y -= hook.speed;
    
    if (hook.target) {
        // Lock fish position to hook
        hook.target.x = hook.x;
        hook.target.y = hook.y + 10;
    }
}
