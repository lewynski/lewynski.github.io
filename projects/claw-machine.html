Cyber Catch

A virtual claw machine featuring a 4-stage state machine (Idle, Drop, Grab, Retract) and proximity-based collision detection.

// --- Logic: The Grab Check ---
// Determines if the claw caught an item

function attemptGrab() {
    let bestDist = 1000;
    let chosenIndex = -1;

    // 1. Iterate through all prizes on floor
    prizes.forEach((p, index) => {
        
        // Calculate distance (Hypotenuse)
        let dx = claw.x - p.x;
        let dy = (claw.y + 20) - p.y;
        let dist = Math.sqrt(dx*dx + dy*dy);

        // 2. Check if close enough
        if (dist < p.type.radius + 15 && 
            dist < bestDist) {
            bestDist = dist;
            chosenIndex = index;
        }
    });

    // 3. Attach Prize to Claw
    if (chosenIndex !== -1) {
        claw.grabbedItem = prizes[chosenIndex];
        
        // Remove from floor array
        prizes.splice(chosenIndex, 1); 
    }

    // 4. Retract upwards
    claw.state = 'RETRACTING';
}
