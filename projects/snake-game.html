Cyber Viper

A modernized snake game with dynamic speed scaling, particle effects, and a "Boost" risk-reward mechanic.

// --- Dynamic Speed & Game Loop ---
// Controls the difficulty ramping and boost mechanic

function gameLoop() {
    if (!isRunning) return;

    update();
    draw();

    // 1. Determine Delay (Speed)
    // If 'isBoosting' is true, cut delay in half (Double Speed)
    let delay = isBoosting ? currentSpeed / 2 : currentSpeed;

    // 2. Recursive Timeout for variable speed
    setTimeout(gameLoop, delay);
}

function update() {
    // ... movement logic ...

    // 3. Difficulty Ramping
    // Every 50 points (5 apples), increase base speed
    if (score % 50 === 0 && baseSpeed > 40) {
        baseSpeed -= 5;
        currentSpeed = baseSpeed;
    }
}
