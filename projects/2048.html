function slide(row) {
    // Filter out zeros to "slide" numbers to the left
    let arr = row.filter(val => val);
    
    // Pad the rest with zeros
    let missing = 4 - arr.length;
    let zeros = Array(missing).fill(0);
    
    return arr.concat(zeros);
}
