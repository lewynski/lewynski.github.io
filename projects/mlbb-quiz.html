Mobile Legends Quiz

Test your game knowledge on the road to Mythic. Features a Mock API implementation to simulate async data loading.

// --- Mock API Simulation ---
// Since MLBB has no public quiz API, we simulate
// a server fetch using Promises and setTimeout.

function fetchQuestionsMock() {
    return new Promise((resolve) => {
        
        // 1. Simulate Network Latency (1.5s)
        setTimeout(() => {
            
            // 2. Shuffle the master database
            const shuffled = [...MLBB_DATA]
                .sort(() => 0.5 - Math.random());

            // 3. Return random subset (10 questions)
            resolve(shuffled.slice(0, 10));
            
        }, 1500);
        
    });
}

// --- Async Game Initialization ---
async function initGame() {
    showLoadingScreen();
    
    // The app waits here, just like a real API call
    const questions = await fetchQuestionsMock();
    
    hideLoadingScreen();
    startGame(questions);
}
