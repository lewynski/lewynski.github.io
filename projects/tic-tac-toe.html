A Tic-Tac-Toe engine powered by the Minimax Algorithm. It recursively calculates every possible future outcome to ensure it never loses.

// --- The Minimax Algorithm ---
// Recursively checks every future board state

function minimax(board, depth, isMaximizing) {
    
    // 1. Base Cases (Game Over conditions)
    if (checkWin(board, AI)) return 10 - depth;
    if (checkWin(board, HUMAN)) return depth - 10;
    if (isDraw(board)) return 0;

    // 2. Recursive Step
    if (isMaximizing) {
        let bestScore = -Infinity;
        for (let i = 0; i < 9; i++) {
            if (!board[i]) {
                board[i] = AI;
                // Call self to check resulting state
                let score = minimax(board, depth + 1, false);
                board[i] = null; // Backtrack
                bestScore = Math.max(score, bestScore);
            }
        }
        return bestScore;
    }
