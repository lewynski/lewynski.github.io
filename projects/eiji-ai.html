Eiji Assistant

A conversational interface powered by simple keyword logic and the browser's native Text-to-Speech API.

// --- Text-to-Speech Logic ---
// Uses the Web Speech API to make Eiji talk

function speak(text) {
    // 1. Check Browser Support
    if (!window.speechSynthesis) return;

    // 2. Create Utterance Object
    const utterance = new SpeechSynthesisUtterance(text);
    
    utterance.pitch = 1; // Normal tone
    utterance.rate = 1;  // Normal speed

    // 3. Handle Events (Visual Feedback)
    utterance.onstart = () => {
        aiRing.classList.remove('hidden'); // Show animation
    };

    utterance.onend = () => {
        aiRing.classList.add('hidden'); // Hide animation
    };

    // 4. Speak!
    window.speechSynthesis.speak(utterance);
}
