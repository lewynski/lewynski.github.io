Neon Flap

A physics-based "Flappy Bird" clone using HTML5 Canvas. Features collision detection, gravity simulation, and score tracking.

// --- Physics Engine Logic ---
// This object controls movement & gravity

const bird = {
    y: 150,      // Vertical Position
    speed: 0,    // Current Velocity
    gravity: 0.25, // Downward Force
    jump: 4.6,   // Upward Force

    // Runs every frame (60 FPS)
    update: function() {
        // 1. Apply Gravity (Accelerate Down)
        this.speed += this.gravity;
        
        // 2. Move Bird Position
        this.y += this.speed;

        // 3. Floor Collision Check
        if(this.y + this.h >= canvas.height) {
            gameOver();
        }
    },

    // Called on Tap / Spacebar
    flap: function() {
        // Set upward velocity instantly
        this.speed = -this.jump;
    }
};
