A complete linear algebra engine capable of solving A+B, A-B, A×B, and A·B⁻¹.

// --- Matrix Multiplication Logic ---
// Performs standard row-by-column dot product

function multiplyMatrices(A, B) {
    const size = A.length;
    
    // Initialize Result Matrix with zeros
    let result = new Array(size)
        .fill(0)
        .map(() => new Array(size).fill(0));

    // O(N³) Complexity Loop
    for (let r = 0; r < size; r++) {
        for (let c = 0; c < size; c++) {
            // Calculate Dot Product for cell (r,c)
            for (let i = 0; i < size; i++) {
                result[r][c] += A[r][i] * B[i][c];
            }
        }
    }
    
    return result;
}

// Note: Division is handled as A * Inverse(B)
// using Gaussian Elimination for inversion.
